82  Double_t LinePointDistance2(Double_t x,Double_t y,Double_t z, const Double_t *lineParams) {
83  // distance line point is D= | (xp-x0) cross ux |
84  // where ux is direction of line and x0 is a point in the line (like t = 0)
85  ROOT::Math::XYZVector vectorPoint(x,y,z);
86  ROOT::Math::XYZVector vectorPointOnLine1(lineParams[0], lineParams[2], 0. );
87  ROOT::Math::XYZVector vectorPointOnLine2(lineParams[0] + lineParams[1], lineParams[2] + lineParams[3], 1. );
88  ROOT::Math::XYZVector lineUnitVector = (vectorPointOnLine2-vectorPointOnLine1).Unit();
89  Double_t distance2 = ((vectorPoint-vectorPointOnLine1).Cross(lineUnitVector)).Mag2();
90  return distance2;
91  }
